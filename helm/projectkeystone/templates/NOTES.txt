ProjectKeystone HMAS has been deployed!

1. Check the deployment status:

  kubectl get pods --namespace {{ include "projectkeystone.namespace" . }} -l "app.kubernetes.io/name={{ include "projectkeystone.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

2. Access the health check endpoint:

{{- if .Values.service.type | eq "ClusterIP" }}

  kubectl port-forward --namespace {{ include "projectkeystone.namespace" . }} svc/{{ include "projectkeystone.fullname" . }} 8080:{{ .Values.service.ports.health.port }}

  Then visit http://127.0.0.1:8080/healthz in your browser.

{{- else if .Values.service.type | eq "NodePort" }}

  export NODE_PORT=$(kubectl get --namespace {{ include "projectkeystone.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "projectkeystone.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ include "projectkeystone.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")

  echo "Health endpoint: http://$NODE_IP:$NODE_PORT/healthz"

{{- else if .Values.service.type | eq "LoadBalancer" }}

  export SERVICE_IP=$(kubectl get svc --namespace {{ include "projectkeystone.namespace" . }} {{ include "projectkeystone.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")

  echo "Health endpoint: http://$SERVICE_IP:{{ .Values.service.ports.health.port }}/healthz"

{{- end }}

3. View the logs:

  kubectl logs --namespace {{ include "projectkeystone.namespace" . }} -f deployment/{{ include "projectkeystone.fullname" . }}

4. Access the Prometheus metrics endpoint:

  kubectl port-forward --namespace {{ include "projectkeystone.namespace" . }} svc/{{ include "projectkeystone.fullname" . }} 9090:{{ .Values.service.ports.metrics.port }}

  Then visit http://127.0.0.1:9090/metrics in your browser.
